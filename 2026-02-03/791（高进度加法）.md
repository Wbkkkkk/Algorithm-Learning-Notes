## 加法模板
```
vector<int> add(vector<int> &a,vector<int> &b){//为啥要用'&'，这样可以省去复制vector内容的一步
    vector<int> c;
    int temp=0;//记录是否进位

    for(int i=0;i<a.size()||i<b.size();i++){
        if(i<a.size())  temp+=a[i];
        if(i<b.size())  temp+=b[i];
        c.push_back(temp%10);
        temp/=10;
    }

    if(temp)  c.push_back(1);//如果最高位还有进位，用1补上

    return c;
}
```
