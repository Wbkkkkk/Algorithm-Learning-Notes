# 快排模板
```
void quicksort(int q[],int l,int r){//把边界传过来

    if(l>=r)  return;//如果只有一个元素，或不合法，直接退出，快排只有这一个等号，并且不能省略

    int m=rand()%(r-l+1)+l;//随机取一个枢轴，但是起始位置要加一个l，而不是1
    swap(q[l],q[m]);//将枢轴换到起始位置
    int x=q[l];
    int i=l-1,j=r+1;
    
    while(i<j){
        do i++;  while(q[i]<x);
        do j--;  while(q[j]>x);
        if(i<j)  swap(q[i],q[j]);
    }

    quicksort(q,l,j);
    quicksort(q,j+1,r);//递归（l...j...i...r）,所以分成l到j，j+1到r
}
```
